conf t
hostname ACCESS2
banner motd # SWITCH ACCESS2 - SOLO ACCESO AUTORIZADO #
ip domain-name arrieta.hn
username admin privilege 15 secret Arrieta2025

! Acceso remoto
line vty 0 15
 login local
 transport input all
line con 0
 login local
exit

! VLANs locales
vlan 300
 name DESARROLLO_DE_SOFTWARE
vlan 400
 name ADMINISTRACION

! No se crean interfaces VLAN porque el gateway lo dan los DIST (HSRP)

! Puertos de usuarios
interface f0/2
 description PC_DESARROLLO_VLAN300
 switchport mode access
 switchport access vlan 300
 spanning-tree portfast
 spanning-tree bpduguard enable
 no shutdown
exit

interface f0/3
 description PC_ADMINISTRACION_VLAN400
 switchport mode access
 switchport access vlan 400
 spanning-tree portfast
 spanning-tree bpduguard enable
 no shutdown
exit

! Doble uplink hacia ambos DIST (redundancia)
interface fa0/1
 
 description TRUNK_HACIA_DIST1
 switchport mode trunk
 switchport trunk allowed vlan 300,400
 no shutdown
exit

interface fa0/5
 description TRUNK_HACIA_DIST2
 switchport mode trunk
 switchport trunk allowed vlan 300,400
 no shutdown
exit

spanning-tree mode rapid-pvst
spanning-tree portfast default
