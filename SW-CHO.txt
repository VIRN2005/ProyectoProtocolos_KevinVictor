!================================================
! SWITCH CHOLUTECA (SW-CHO)
!================================================
en
conf t
hostname SW-CHO
banner motd # GRUPO ARRIETA CHOLUTECA SWITCH - SOLO ACCESO AUTORIZADO #
ip domain-name arrieta.hn
username admin privilege 15 secret Arrieta2025

! Acceso remoto y consola
line vty 0 15
 login local
 transport input all
line con 0
 login local
exit

! Configuración de VLANs
vlan 100
 name LOGISTICA
exit
vlan 200
 name SOPORTE_TECNICO
exit
vlan 300
 name DESARROLLO_DE_SOFTWARE
exit
vlan 400
 name ADMINISTRACION
exit

! Configuración de trunk al router
interface GigabitEthernet0/1
 description ENLACE-ROUTER-CHOLUTECA
 switchport mode trunk
 switchport trunk allowed vlan 100,200,300,400
 switchport trunk native vlan 999
 no shutdown
exit

! Puertos de acceso VLAN 100 (FINANZAS)
interface range FastEthernet0/1-5
 description FINANLOGISTICAZAS-VLAN100
 switchport mode access
 switchport access vlan 100
 switchport port-security
 switchport port-security maximum 2
 switchport port-security mac-address sticky
 switchport port-security violation shutdown
 spanning-tree portfast
 no shutdown
exit

! Puertos de acceso VLAN 200 (SERVICIO AL CLIENTE)
interface range FastEthernet0/6-10
 description SOPORTE_TECNICO-VLAN200
 switchport mode access
 switchport access vlan 200
 switchport port-security
 switchport port-security maximum 2
 switchport port-security mac-address sticky
 switchport port-security violation shutdown
 spanning-tree portfast
 no shutdown
exit

! Puertos de acceso VLAN 300 (SISTEMAS)
interface range FastEthernet0/11-15
 description DESARROLLO_DE_SOFTWARE-VLAN300
 switchport mode access
 switchport access vlan 300
 switchport port-security
 switchport port-security maximum 2
 switchport port-security mac-address sticky
 switchport port-security violation shutdown
 spanning-tree portfast
 no shutdown
exit

! Puertos de acceso VLAN 400 (ADMINISTRACION)
interface range FastEthernet0/16-20
 description ADMINISTRACION-VLAN400
 switchport mode access
 switchport access vlan 400
 switchport port-security
 switchport port-security maximum 2
 switchport port-security mac-address sticky
 switchport port-security violation shutdown
 spanning-tree portfast
 no shutdown
exit

! SVI para administración
interface vlan 300
 ip address 192.168.2.130 255.255.255.192
 ipv6 address 2001:db8:3000:2::130/64
 no shutdown
exit

! Gateway por defecto
ip default-gateway 192.168.2.129

end
copy running-config startup-config
